CREATE TABLE PLAYER(
id BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(45) NOT NULL UNIQUE ,
password VARCHAR(45) NOT NULL,
X INT NOT NULL DEFAULT 832,
Y INT NOT NULL DEFAULT 1248,
desiredX INT NOT NULL DEFAULT 832,
desiredY INT NOT NULL DEFAULT 1248,
HP INT NOT NULL DEFAULT 100,
direction char NOT NULL DEFAULT 'd'
outfit VARCHAR(45) NOT NULL DEFAULT 'GM'
);
